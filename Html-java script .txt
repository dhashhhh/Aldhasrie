  <script>
    document.addEventListener("DOMContentLoaded", function() {
      const links = document.querySelectorAll(".navbar-links a");
      const content = document.querySelector(".container");
      
      links.forEach(function(link) {
        link.addEventListener("click", function(e) {
          e.preventDefault();
          const page = this.getAttribute("href");
          updateContent(page);
        });
      });
      
      function updateContent(page) {
        // Replace this section with your own logic to update the content
        if (page === "home.html") {
          content.innerHTML = "<h1>Welcome to the Home Page</h1><p>This is the home page content.</p>";
        } else if (page === "about.html") {
          content.innerHTML = "<h1>About Us</h1><p>About page content goes here.</p>";
        } else if (page === "services.html") {
          content.innerHTML = "<h1>Our Services</h1><p>Here are the services we provide.</p>";
        } else if (page === "contact.html") {
          content.innerHTML = "<h1>Contact Us</h1><p>Contact information and form will be displayed here.</p>";
        }
      }
    });
  </script>
